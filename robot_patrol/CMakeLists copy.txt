cmake_minimum_required(VERSION 3.5)
project(robot_patrol)

# Default to C99
if(NOT CMAKE_C_STANDARD)
  set(CMAKE_C_STANDARD 99)
endif()

# Default to C++14
if(NOT CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 14)
endif()

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# find dependencies
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(nav_msgs REQUIRED)
find_package(geometry_msgs REQUIRED)
find_package(rosidl_default_generators REQUIRED)
find_package(std_srvs REQUIRED)

if(BUILD_TESTING)
  find_package(ament_lint_auto REQUIRED)
  
  ament_lint_auto_find_test_dependencies()
endif()

rosidl_generate_interfaces(${PROJECT_NAME}
	"srv/GetDirection.srv"
	DEPENDENCIES sensor_msgs std_srvs
)

add_executable(patrol_node src/patrol.cpp)
ament_target_dependencies(patrol_node rclcpp nav_msgs geometry_msgs sensor_msgs)

add_executable(direction_service_node src/direction_service.cpp)
ament_target_dependencies(direction_service_node rclcpp nav_msgs geometry_msgs sensor_msgs)

add_executable(test_node src/test_service.cpp)
ament_target_dependencies(test_node rclcpp nav_msgs geometry_msgs sensor_msgs)

install(TARGETS
	patrol_node
	direction_service_node
	test_node
	DESTINATION lib/${PROJECT_NAME}/
	)

install(DIRECTORY
  launch
  DESTINATION share/${PROJECT_NAME}/
  )
  
ament_package()
